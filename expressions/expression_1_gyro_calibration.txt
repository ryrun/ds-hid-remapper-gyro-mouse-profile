/* Inspired by Alpakka firmware by Input Labs: https://inputlabs.io/ */
/* Purpose: map gyro movement to relative mouse X/Y with touch gating, bias auto-calibration, and a soft curve. */
/* Mappable registers: R1 = mouse X delta, R2 = mouse Y delta. */
/* Temp registers: R19 = stillness flag, R20 = touch active, R27/R28 = gyro bias X/Y, R29/R30 = raw gyro X/Y, R31/R32 = filtered X/Y. */

/* Touch is active when at least one touch contact is present. */
0xfffa0002 input_state not
0xfffa0006 input_state not
bitwise_or
20 store

/* Read raw gyro axes used for mouse X/Y. */
0xfffa000a input_state
29 store
0xfffa0009 input_state
30 store

/* Only calibrate while untouched and nearly still on both axes. */
29 recall abs 12 lt
30 recall abs 12 lt
mul
20 recall not
mul
19 store

/* Slowly adapt X bias toward the current raw value while still. */
19 recall
27 recall 29 recall 27 recall sub 0.01 mul add
27 recall
ifte
27 store

/* Slowly adapt Y bias toward the current raw value while still. */
19 recall
28 recall 30 recall 28 recall sub 0.01 mul add
28 recall
ifte
28 store

/* Read gyro X, subtract bias, then suppress tiny residuals smoothly. */
29 recall 27 recall sub
31 store
31 recall abs 15 lt
31 recall 31 recall abs 15 div mul
31 recall
ifte

/* Apply base sensitivity and invert direction. */
-0.004 mul

/* Require at least one active touch contact. */
20 recall
mul

/* Keep scaled X in temp register 31. */
31 store

/* Condition: use soft curve only for |x| < 1. */
31 recall abs 1 lt

/* Soft shaping: sign(x) * (0.5 * |x|) / (1 - 0.5 * |x|) */
31 recall sign 31 recall abs 0.5 mul 1 31 recall abs 0.5 mul sub div mul

/* Linear fallback for |x| >= 1. */
31 recall

/* Choose shaped value for small inputs, linear value otherwise. */
ifte

/* Store final mouse X delta in register 1. */
1 store

/* Read gyro Y, subtract bias, then suppress tiny residuals smoothly. */
30 recall 28 recall sub
32 store
32 recall abs 15 lt
32 recall 32 recall abs 15 div mul
32 recall
ifte

/* Apply base sensitivity and invert direction. */
-0.004 mul

/* Require at least one active touch contact. */
20 recall
mul

/* Keep scaled Y in temp register 32. */
32 store

/* Condition: use soft curve only for |y| < 1. */
32 recall abs 1 lt

/* Soft shaping: sign(y) * (0.5 * |y|) / (1 - 0.5 * |y|) */
32 recall sign 32 recall abs 0.5 mul 1 32 recall abs 0.5 mul sub div mul

/* Linear fallback for |y| >= 1. */
32 recall

/* Choose shaped value for small inputs, linear value otherwise. */
ifte

/* Store final mouse Y delta in register 2. */
2 store

/* Dummy expression result. */
0
